SELECT DISTINCT idcase FROM event_log WHERE activity IN ( SELECT activity FROM event_log GROUP BY activity HAVING COUNT(DISTINCT timestamp) > 1 ) GROUP BY idcase HAVING COUNT(DISTINCT activity) > 1;
SELECT activity, timestamp, resource, idcase FROM event_log WHERE activity = (SELECT activity FROM event_log GROUP BY activity HAVING COUNT(DISTINCT timestamp) > 1) ORDER BY timestamp;
